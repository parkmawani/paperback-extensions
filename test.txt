            jQuery('.go-top:first').after('<span class="go-comment cursor"><i class="fa fa-commenting"></i></span>')
        }

        jQuery(".go-comment").on('click', function() {
            // $('html, body').animate({ scrollTop: jQuery(document).height() }, 500);
            var toComment = jQuery('#viewcomment').offset();
            if( toComment != undefined && toComment['top'] > 0 ) {
                var toPosition = toComment['top'] - 60;
            } else {
                var toPosition = jQuery(document).height();
            }
            $('html, body').scrollTop(toPosition);
            return false;
        });

        jQuery("header.main-header").sticky({topSpacing:0, zIndex: 1030});
        jQuery("section.sidebar").sticky({topSpacing:50, getWidthFrom: '.main-sidebar'});
        jQuery(".control-sidebar-content").sticky({topSpacing:50, getWidthFrom: '.control-sidebar', zIndex: 1030});

        jQuery('.to-show').on('click', function () {
            var is_visible = jQuery('#to-menu').is(":visible");
            if(!is_visible){
                jQuery('#to-btn span').css({
                    '-webkit-transform': 'rotate(45deg)',
                    '-moz-transform': 'rotate(45deg)',
                    'transform': 'rotate(45deg)',
                    'background': 'rgba(193,9,9,0.5)',
                    'transition' : '.3s'
                });
                jQuery("#to-menu" ).show();

                // jQuery('#to-menu').show();
                // jQuery('#to-btn span i').addClass('fa-times').removeClass('fa-plus');
            } else {
                jQuery('#to-menu').hide();
                jQuery('#to-btn span').css({
                    '-webkit-transform': 'rotate(0deg)',
                    '-moz-transform': 'rotate(0deg)',
                    'transform': 'rotate(0deg)',
                    'background': 'rgba(0,0,0,0.3)',
                    'transition' : '.3s'
                });
                jQuery('#to-btn span i').addClass('fa-plus').removeClass('fa-times');
            }
        });

    });
